open Dictionary

module Make =
functor
  (K : KeySig)
  (V : ValueSig)
  ->
  struct
    module Key = K
    module Value = V

    type key = K.t
    type value = V.t
    let cmp = K.compare

    (** [color] represents the color of a tree node. *)
    type color =
      | Black
      | Red

    type rbtree =
      | Leaf
      | Node of color * key * value * rbtree * rbtree

    (** [rbtree] is a red-black tree with a key and value for every node.

        AF: Each value [t] that is type [rbtree] is a dictionary. 
        If t is a [Leaf] then it has no bindings, else if it is a 
        [Node (color, k, v, left, right)] then it contains a binding [k] to 
        [v] in addition to all the bindings contained in [left] and [right].

        RI 1: BST Invariant
        (No identical values, all nodes to the left of a 
        node have lower keys and all nodes to the right have higher keys)

        RI 2: Red-black local invariant
        (A red node can't have any direct red node children.)

        RI 3: Red-black globla invariant
        (Every path from the root to a leaf has an identical number of
        black nodes with leaves being considered as black nodes.) *)

    type t = rbtree

    (* [lt_all] is whether all the nodes to the left of a [rbtree] have 
    values less than itself. Returning the appropriate boolean. *)
    let rec lt_all value = function
    | Leaf -> true
    | Node (_, k, _, l, r) ->
        Key.compare k value = LT && lt_all value l && lt_all value r
    [@@coverage off]

    (* [gt_all] is whether all the nodes to the right of a [rbtree] have 
    values greater than itself. Returning the appropriate boolean.*)
    let rec gt_all value = function
      | Leaf -> true
      | Node (_, k, _, l, r) ->
          Key.compare k value = GT && gt_all value l && gt_all value r
      [@@coverage off]
    (* [valid_bst] is whether the [rbtree] t upholds the BST invariant using 
       calls to helper functions [lt_all] and [gt_all]*)
    let rec valid_bst t =
      match t with
      | Leaf -> true
      | Node (_, k, v, l, r) ->
          lt_all k l && gt_all k r && valid_bst l && valid_bst r
      [@@coverage off]
    
    (* [rb_local] is whether the [rbtree] violates the local invariant 
    of not having a red parent and corresponding red child. Returns boolean. *)
    let rec rb_local parent_c = function
      | Leaf -> true
      | Node (Black, _, _, l, r) -> 
        rb_local Black l && rb_local Black r
      | Node (Red, _, _, left, right) ->
        parent_c = Black && rb_local Red left && rb_local Red right
      [@@coverage off]
    
    (* [rb_glob] is whether the [rbtree] violates the global invariant of
    having an equal number of black nodes for every path *)
    let rec rb_glob n = function
      | Leaf -> n = 0
      | Node (Black, _, _, left, right) ->
        rb_glob (n - 1) left && rb_glob (n - 1) right
      | Node (Red, _, _, left, right) ->
        rb_glob n left && rb_glob n right
      [@@coverage off]

    (* [ref_b_height] is the number of black nodes along the left-most
    path of the [rbtree] to reference in [rb_glob]*)
    let rec ref_b_height count = function
      | Leaf -> count
      | Node (Black, _, _, left, _) -> ref_b_height (count + 1) left
      | Node (Red, _, _, left, _) -> ref_b_height count left
      [@@coverage off]
      
    (* set to true when debugging for rep_ok *)
    let debug = false
    
    let rep_ok d = 
      if debug then 
        if valid_bst d && rb_local Black d 
        && rb_glob (ref_b_height 0 d) d then d 
          else failwith "RI violated"
      else d
      [@@coverage off]

    let empty = rep_ok Leaf

    let is_empty d = rep_ok Leaf = d

    let rec size d =
      match rep_ok d with
      | Leaf -> 0
      | Node (_, _, _, l, r) -> 1 + size l + size r
    
    (* [balance] returns the appropriate Okasaki tree balancing rotation *)
    let balance = function
      | Black, kz, vz, Node (Red, ky, vy, Node (Red, kx, vx, a, b), c), d
      | Black, kz, vz, Node (Red, kx, vx, a, Node (Red, ky, vy, b, c)), d
      | Black, kx, vx, a, Node (Red, kz, vz, Node (Red, ky, vy, b, c), d)
      | Black, kx, vx, a, Node (Red, ky, vy, b, Node (Red, kz, vz, c, d)) ->
        Node (Red, ky, vy,  Node (Black, kx, vx, a, b), 
        Node (Black, kz, vz, c, d))
      | a, b, c, d, e -> Node (a, b, c, d, e)

    (* [insert_helper] is the appropriately balanced [rbtree] of [d] after
    a newly inserted node with key [k] and value [v] *)
    let rec insert_helper k v d = 
      match d with
      | Leaf -> Node (Red, k, v, Leaf, Leaf)
      | Node (color, k', v', a, b) ->
          if Key.compare k k' = LT then 
            balance (color, k', v', insert_helper k v a, b)
          else if Key.compare k k' = GT 
            then balance (color, k', v', a, insert_helper k v b)
          else Node (color, k, v, a, b)
    
    (* [insert] executes an insertion of a node with key [k] and value [v] into 
    [rbtree] [d] using [insert_helper] and switches the root to black*)
    let insert k v d =
      match insert_helper k v (rep_ok d) with
      | Node (_, k', v', l, r) -> Node (Black, k', v', l, r)
      | Leaf -> failwith "Insert_Called_On_Leaf_Error"

    let remove k d = raise (Failure "Unimplemented: TreeDictionary.Make.remove")
    [@@coverage off]

    let rec find k d = 
      match rep_ok d with
      | Leaf -> None
      | Node (_, k', v, l, r) -> (
          match cmp k k' with
          | LT -> find k l
          | EQ -> Some v
          | GT -> find k r)

    let member k d = 
      match find k d with
      | Some _ -> true
      | None -> false

    let rec to_list d = 
      match d with
      | Leaf -> []
      | Node (_, k, v, l, r) -> to_list l @ [ (k, v) ] @ to_list r

    let rec fold f acc d = 
      match rep_ok d with
      | Leaf -> acc
      | Node (_, k, v, l, r) ->
          let l_acc = fold f acc l in
          let acc2 = f k v l_acc in
          fold f acc2 r

    let to_string d =
      Util.string_of_bindings Key.to_string Value.to_string (to_list d)
  end
